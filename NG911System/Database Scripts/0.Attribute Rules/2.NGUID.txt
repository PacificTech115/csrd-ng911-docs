var ag = Trim(DefaultValue($feature.DiscrpAgID, ""));
if (ag == "") return Null;

// Use the synced Municipal ID (FeatureID) if available, otherwise generate a new Central ID
var gid = DefaultValue($feature.FeatureID, $feature.GlobalID);

// GlobalID sometimes comes with { } — strip them
gid = Replace(Replace(Text(gid), "{", ""), "}", "");

return "urn:emergency:uid:gis:SSAP:" + gid + ":" + ag;
